def bubble_sort(a, N):
    for i in range(1,N): # sort할 마지막 원소의 인덱스 i
        for j in range(0, N-i): # N-i까지의 원소를 bubble sort할 것임.
            if a[j] > a[j+1]:
                a[j], a[j+1] = a[j+1], a[j]
    return a
